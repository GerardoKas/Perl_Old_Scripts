$dir=shift @ARGV;
print "Listado de \"$dir\"";
opendir(DR,$dir) or die "NO EXISTE $dir";
@fs=readdir(DR);
closedir DR;

@fs=grep {!/^\./} @fs;

print ($#fs+1)." Archivos";

@fs=map{$_="\"$_\""} @fs;

$lista=join(", ",@fs);

open(FL,">listado.txt") or die "ERROR EsCRITURA";
print FL $lista;
close FL;

print "HECHO\n";
system "pause";
